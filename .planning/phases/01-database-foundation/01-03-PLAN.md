For the watchlist add/remove propagation to work, WatchlistRepository.add/remove need to also update the live PriceCache and MarketDataSource. Implement this in `backend/app/main.py` (or a helper module) by creating two thin sync-to-cache functions:

```python
async def add_ticker_with_sync(ticker: str) -> dict:
    """Add ticker to DB, then sync to PriceCache and MarketDataSource."""
    repo = WatchlistRepository()
    row = await repo.add(ticker)
    if state.market_source is not None:
        await state.market_source.add_ticker(row['ticker'])
    return row


async def remove_ticker_with_sync(ticker: str) -> bool:
    """Remove ticker from DB, then sync from PriceCache and MarketDataSource."""
    repo = WatchlistRepository()
    removed = await repo.remove(ticker)
    if removed:
        if state.market_source is not None:
            await state.market_source.remove_ticker(ticker.upper())
        if state.price_cache is not None:
            state.price_cache.remove(ticker.upper())
    return removed
```
